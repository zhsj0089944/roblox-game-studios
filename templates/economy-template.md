### DataStore 存储

```lua
-- 经济数据结构
local economyData = {
    score = 0,           -- 当前积分
    totalScore = 0,      -- 总积分（含已消耗）
    ep = 0,              -- 当前 EP
    totalEP = 0,         -- 总 EP
    upgrades = {         -- 升级等级
        range = 1,
        multiplier = 1,
        field = 1,
        launch = 1,
    },
    evolutions = {       -- 进化等级
        finger = 0,
        field = 0,

        critRate = 0,
        dodgeRate = 0,

```

### 公式实现
```lua
-- 积分计算公式
local function calculateScore(data, object, eventMult)
    local base = object.baseScore or 1
    local itemBonus = object.itemBonus or 0
    local multUp = getUpgradeMultiplier(data.upgrades.multiplier)
    local evoTouch = getEvolutionBonus(data.evolutions.finger)
    local vipMult = data.isVIP and 1.5 or 1.0

    return math.floor(
        base 
        * (1 + itemBonus) 
        * (1 + multUp) 
        * (1 + evoTouch) 
        * eventMult 
        * vipMult
    )
end

-- EP 计算公式（炼化时）
local function calculateEP(totalScore, totalResets)
    local epRate = 0.001
    local cap = 20
    local effectiveResets = math.min(totalResets, cap)
    
    return math.floor(totalScore * epRate * (1 + effectiveResets * 0.1))
end
```

### 防作弊
- [所有经济计算在服务器]
- [客户端不信任任何数值]
- [DataStore 操作有重试和错误处理]
- [关键操作有日志记录]

### 性能考虑
- [积分计算在热路径中，避免内存分配]
- [DataStore 请求有频率限制]
- [大量数据使用 OrderedDataStore]
